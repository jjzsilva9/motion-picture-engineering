% Slide 2: The Matting Problem
\begin{frame}{Problem Formulation}
    \textbf{Composition Equation:}
    \[ I = \alpha F + (1-\alpha)B \]
    \vfill
    \textbf{Energy Minimization:}
    \[ \hat{\alpha} = \arg\min [-\ln P(I | \alpha) - \ln P(\alpha)] \]
\end{frame}

% Slide 3: Likelihood Model
\begin{frame}{Likelihood}
    \[ E_{likelihood}(c | \alpha=0) = \frac{1}{2} \sum_{k \in \{R,G,B\}} \frac{(c_k - \mu_k)^2}{\sigma_k^2} \]
    
    \begin{itemize}
        \item \textbf{$c$: source image}
        \item \textbf{$\mu$: mean of background}
        \item \textbf{$\sigma^2$: variance of background}
    \end{itemize}
\end{frame}

% Slide 4: Spatial Prior (MRF)
\begin{frame}{Prior}
    \[ E_{prior}(\alpha_p) = \Lambda \sum_{q \in \mathcal{N}_p} w_{pq} \cdot ||\alpha_p \neq \alpha_q|| \]
    
    \[ w_{pq} = \exp\left( - \frac{d(p,q)^2}{2\sigma_{spatial}^2} \right) \]
    
    \begin{itemize}
        \item Encourages neighboring pixels to have same labels.
        \item \textbf{Hyperparameters:} $\mathcal{N}_p$ (number of neighbours), $\Lambda$ (prior strength) and $\sigma_{spatial}$ (per-neighbour weighting)
    \end{itemize}
\end{frame}

% Slide 5: Implementation - BlinkScript & ICM
\begin{frame}{Implementation: Optimizing the Posterior}
    \begin{enumerate}
        \item Use MLE (likelihood only) as the starting matte.
        \item For each pixel $p$, update $\alpha_p$ to minimize total energy:
        \[ \alpha_t[p] = \arg\min_{\alpha \in \{0,1\}} (E_{likelihood} + E_{prior}(\alpha, \alpha_{t-1}[\mathcal{N}_p])) \]
    \end{enumerate}
    \vfill
    \centering
    \fbox{\footnotesize \texttt{Value = (Energy\_FG < Energy\_BG) ? 1.0f : 0.0f;}}
\end{frame}

% Slide 10: Shortcomings & Future Directions
\begin{frame}{Shortcomings and Improvements}
    \textbf{Current Shortcomings:}
    \begin{itemize}
        \item \textbf{Binary Constraint:} Cannot represent sub-pixel details (soft hair).
        \item \textbf{Greedy Optimization:} ICM can get trapped in local energy minima.
        \item \textbf{Iteration Count:} Requires multiple passes for full convergence.
    \end{itemize}
    \vfill
    \textbf{Future Improvements:}
    \begin{itemize}
        \item \textbf{Soft Matting:} Generalize $\alpha \in [0, 1]$ using a continuous MRF or Matting Laplacian.
        \item \textbf{Temporal Consistency:} Extend neighborhood $\mathcal{N}$ to 3D (spatial + time) for flicker reduction.
    \end{itemize}
\end{frame}
